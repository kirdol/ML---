Code 
#Number of occurences/model per make 
nb_model_per_make <- data %>%
  group_by(Make, Model) %>%
  summarise(Number = n(), .groups = 'drop') %>%
  group_by(Make) %>%
  summarise(Models_Per_Make = n(), .groups = 'drop') %>%
  arrange(desc(Models_Per_Make))
